x = np.linspace(-4*np.pi,4*np.pi,501)
def plot(r):
    fig, ax = plt.subplots(figsize=(4, 3),
                           subplot_kw={'axisbg':'#EEEEEE',
                                       'axisbelow':True})
    ax.grid(color='w', linewidth=2, linestyle='solid')
    F = 4*r/(1-r)**2
    ax.plot(x,1/(1+F*np.sin(x/2)**2), lw=5, alpha=0.4)
    #ax.set_xlim(-50, 50)
    ax.set_ylim(0, 1)
    return fig
